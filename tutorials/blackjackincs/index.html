Blackjack in C#

Some of you from Quora know that I am very pro-C#. But what has C# got that Java, C and C++ haven’t?

I guess the simplest possible answer is that C# is programmer-friendlier than the above three combined, carries most of their advantages and none of their disadvantages.

More importantly, C#, now in version 12 (as of November 2023) has many new goodies that will make development in C# pure joy.

If you know some Python, I invite you to take a look at this simple game of Blackjack I wrote in C#:

(If you want to download and try the game on your computer, head on to my github.)

 79                {
 80                    dealerHand.AddCard(deck.DrawCard());
 81                }

 84
 85                if (dealerHand.Value > 21)
 86                {
 87                    Console.WriteLine("Dealer busted! You win!");
 88                }
 89                else if (dealerHand.Value > playerHand.Value)
 90                {
 91                    Console.WriteLine("Dealer wins!");
 92                }
 93                else if (dealerHand.Value < playerHand.Value)
 94                {
 95                    Console.WriteLine("You win!");
 96                }
 97                else
 98                {
 99                    Console.WriteLine("It's a tie!");
100                }
101
102                break;
103            }
104        }

108public class Card
109{
110    public int Value { get; set; }
111    public string Name { get; set; }
112
113    public override string ToString()
114    {
115        return Name;
116    }
117}
118
119public class Deck
120{
121    private List<Card> cards;
122
123    public Deck()
124    {
125        cards = new List<Card>();
126
127        for (int i = 2; i <= 11; i++)
128        {
129            for (int j = 0; j < 4; j++)
130            {
131                var card = new Card { Value = i };
132
133                if (i == 11)
134                {
135                    card.Name = "Ace";
136                    card.Value = 11;
137                }
138                else if (i == 10)
139                {
140                    card.Name = j == 0 ? "10" : j == 1 ? "Jack" : j == 2 ? "Queen" : "King";
141                }
142                else
143                {
144                    card.Name = i.ToString();
145                }
146
147                cards.Add(card);
148            }
149        }
150
151        var random = new Random();
152        for (int i = 0; i < cards.Count; i++)
153        {
154            var temp = cards[i];
155            var index = random.Next(cards.Count);
156            cards[i] = cards[index];
157            cards[index] = temp;
158        }
159    }
160
161    public Card DrawCard()
162    {
163        var card = cards[0];
164        cards.RemoveAt(0);
165        return card;
166    }
167}

169public class Hand
170{
171    private List<Card> cards = new List<Card>();
172
173    public int Value
174    {
175        get
176        {
177            var value = 0;
178            var aces = 0;
179
180            foreach (var card in cards)
181            {
182                value += card.Value;
183                if (card.Name == "Ace")
184                {
185                    aces++;
186                }
187            }
188
189            while (value > 21 && aces > 0)
190            {
191                value -= 10;
192                aces--;
193            }
194
195            return value;
196        }
197    }
198
199    public void AddCard(Card card)
200    {
201        cards.Add(card);
202    }
203
204    public override string ToString()
205    {
206        return string.Join(", ", cards) + " (" + Value + ")";
207    }
208}
